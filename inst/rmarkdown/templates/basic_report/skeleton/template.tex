\newcommand{\customtitle}[3]{%
  \begin{titlepage}
    \begin{center}
  \begin{minipage}{0.15\textwidth}%
      % Logo placement (conditionally include if the path is provided)
      % Check if logo path is provided and exists
    \ifx\relax#1\relax
    % If logo path is empty, do nothing
    \hfill
    \else
      \IfFileExists{#1}{%
        \includegraphics[width=0.8\textwidth]{#1} %
      }{\hfill} % If the logo file does not exist or path is empty, do nothing
    \fi
  \end{minipage}\hspace{10pt}
  \begin{minipage}{0.8\textwidth}%
    \raggedleft\bfseries%
    {{#2}} \\
    \doublerule  % <====================================================
    {#3}\\%
    {\today}\\
  \end{minipage}%
   \end{center}
    \vspace{1cm}
  \end{titlepage}
}
